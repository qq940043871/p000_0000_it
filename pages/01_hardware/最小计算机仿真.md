function loadProgram(idx) {
  resetCPU();
  currentProg = idx;
  RAM.fill(0);
  let addr = 0;
  const prog = PROGRAMS[idx];
  for (const instr of prog.code) {
    RAM[addr++] = instr.op;
    if (instr.arg !== null) RAM[addr++] = instr.arg & 0xFF;
  }
  renderMemory();
  renderProgList();
  log(`加载程序: ${prog.name}`, 'ok');
  log(`写入 ${addr} 字节到 RAM[0x00..0x${(addr-1).toString(16).padStart(2,'0')}]`, 'mem');
}

function stepOnce() {
  if (cpu.halted) { log('CPU 已停机，请复位', 'warn'); return; }

  // 时钟闪烁
  clockTick = !clockTick;
  document.getElementById('clkDot').className = 'clk-dot' + (clockTick ? ' tick' : '');
  cpu.cycles++;
  document.getElementById('cycleCount').textContent = cpu.cycles + ' cycles';

  // 取指 (Fetch)
  cpu.MAR = cpu.PC;
  cpu.IR = RAM[cpu.PC];
  cpu.lastReg = 'IR';
  updateBus(cpu.PC, cpu.IR, 'READ');

  const opcode = cpu.IR;
  cpu.PC++;

  // 译码 & 执行 (Decode & Execute)
  let arg = 0;
  if (needsArg(opcode)) {
    arg = RAM[cpu.PC];
    cpu.PC++;
  }

  executeOp(opcode, arg);
  cpu.cycles++;

  renderAll();
  renderProgList();
}

function needsArg(op) {
  return [OP.LDA_IMM, OP.LDB_IMM, OP.STA, OP.LDA_MEM, OP.JMP, OP.JZ, OP.JNZ, OP.JC].includes(op);
}

function renderProgList() {
  // rebuild prog list with current PC highlighted
  const list = document.getElementById('progList');
  if (!list) return;
  const prog = PROGRAMS[currentProg];
  let addr = 0;
  list.innerHTML = '';
  for (const instr of prog.code) {
    const line = document.createElement('div');
    const isCurrent = addr === cpu.PC || (addr  cpu.PC);
    line.className = 'prog-line' + (isCurrent ? ' current' : '');
    const parts = instr.asm.split(' ');
    line.innerHTML = `${hex(addr)}${hex(instr.op)}${parts[0]}${parts[1] ? ' '+parts[1]+'' : ''}`;
    list.appendChild(line);
    addr += instr.arg !== null ? 2 : 1;
  }
}

function updateBus(addr, data, ctrl) {
  // Address bus (8 bits)
  const ab = document.getElementById('addrBus');
  ab.innerHTML = '';
  for (let i = 7; i >= 0; i--) {
    const b = document.createElement('div');
    b.className = 'bus-bit' + ((addr >> i) & 1 ? ' hi' : '');
    ab.appendChild(b);
  }
  document.getElementById('addrVal').textContent = '0x'+hex(addr);

  // Data bus (8 bits)
  const db = document.getElementById('dataBus');
  db.innerHTML = '';
  for (let i = 7; i >= 0; i--) {
    const b = document.createElement('div');
    b.className = 'bus-bit' + ((data >> i) & 1 ? ' hi' : '');
    db.appendChild(b);
  }
  document.getElementById('dataVal').textContent = '0x'+hex(data);

  // Control bus
  const cb = document.getElementById('ctrlBus');
  cb.innerHTML = '';
  const ctrlMap = { 'READ': 0b00000001, 'WRITE': 0b00000010, 'ALU': 0b00000100, 'IDLE': 0 };
  const ctrlVal = ctrlMap[ctrl] || 0;
  for (let i = 7; i >= 0; i--) {
    const b = document.createElement('div');
    b.className = 'bus-bit' + ((ctrlVal >> i) & 1 ? ' hi' : '');
    cb.appendChild(b);
  }
  document.getElementById('ctrlVal').textContent = ctrl;
}

function drawBus(x1, y1, x2, y2, color, label, active, bidirectional) {
  ctx.strokeStyle = active ? color : `${color}30`;
  ctx.lineWidth = active ? 2 : 1;
  if (active) { ctx.shadowColor = color; ctx.shadowBlur = 4; }
  ctx.beginPath();
  ctx.moveTo(x1, y1);
  // Orthogonal routing
  const mx = (x1 + x2) / 2;
  ctx.lineTo(mx, y1);
  ctx.lineTo(mx, y2);
  ctx.lineTo(x2, y2);
  ctx.stroke();
  ctx.shadowBlur = 0;

  // Arrow
  if (active) {
    drawArrow(mx, y2, x2, y2, color);
    if (bidirectional) drawArrow(mx, y1, x1, y1, color);
  }

  // Label
  if (label) {
    ctx.font = '8px JetBrains Mono, monospace';
    ctx.fillStyle = active ? color : `${color}50`;
    ctx.textAlign = 'center';
    ctx.fillText(label, mx, Math.min(y1, y2) - 4);
  }
}

function drawArrow(fromX, fromY, toX, toY, color) {
  const angle = Math.atan2(toY - fromY, toX - fromX);
  const len = 8;
  ctx.fillStyle = color;
  ctx.beginPath();
  ctx.moveTo(toX, toY);
  ctx.lineTo(toX - len * Math.cos(angle - 0.4), toY - len * Math.sin(angle - 0.4));
  ctx.lineTo(toX - len * Math.cos(angle + 0.4), toY - len * Math.sin(angle + 0.4));
  ctx.closePath();
  ctx.fill();
}

function animateDot(x1, y1, x2, y2, color) {
  // Simple static dot at midpoint for now
  const mx = (x1 + x2) / 2;
  ctx.fillStyle = color;
  ctx.shadowColor = color;
  ctx.shadowBlur = 10;
  ctx.beginPath();
  ctx.arc(mx, y1, 4, 0, Math.PI * 2);
  ctx.fill();
  ctx.shadowBlur = 0;
}

function toggleRun() {
  if (cpu.halted) { log('CPU 已停机，请先复位', 'warn'); return; }
  isRunning = !isRunning;
  const btn = document.getElementById('btnRun');
  if (isRunning) {
    btn.textContent = '⏸ 暂停';
    btn.className = 'btn btn-run running';
    runInterval = setInterval(() => { stepOnce(); drawCircuit(); }, 1000 / clockHz);
    log('CPU 开始运行...', 'ok');
  } else {
    btn.textContent = '▶ 运行';
    btn.className = 'btn btn-run';
    clearInterval(runInterval);
    log('CPU 暂停', 'warn');
  }
}

  renderMemory();
  renderProgList();
  drawCircuit();
  log('CPU 复位完成', 'ok');
}

function updateSpeed() {
  clockHz = parseInt(document.getElementById('speedSlider').value);
  document.getElementById('speedLabel').textContent = clockHz + ' Hz';
  if (isRunning) {
    clearInterval(runInterval);
    runInterval = setInterval(() => { stepOnce(); drawCircuit(); }, 1000 / clockHz);
  }
}

let logCount = 0;
function log(msg, type = 'info') {
  const con = document.getElementById('console');
  const line = document.createElement('div');
  line.className = `log-line ${type}`;
  line.innerHTML = `[${String(logCount++).padStart(4,'0')}]${msg}`;
  con.appendChild(line);
  con.scrollTop = con.scrollHeight;
  if (con.children.length > 200) con.removeChild(con.children[1]);
}
