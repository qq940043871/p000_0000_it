## 交易概述

以太坊交易是在以太坊网络上执行的操作，包括转账以太币、部署智能合约和调用智能合约。本教程将深入分析以太坊交易的源码实现。

## 交易结构

### 交易数据结构

在以太坊源码中，交易的数据结构定义在`core/types/transaction.go`文件中：

```go
// Transaction is an Ethereum transaction.
type Transaction struct {
    data txdata
    // caches
    hash atomic.Value
    size atomic.Value
    from atomic.Value
}

// txdata is the underlying data of a transaction.
type txdata struct {
    AccountNonce uint64          `json:"nonce"    gencodec:"required"`
    Price        *big.Int        `json:"gasPrice" gencodec:"required"`
    GasLimit     uint64          `json:"gas"      gencodec:"required"`
    Recipient    *common.Address `json:"to"       rlp:"nil"` // nil means contract creation
    Amount       *big.Int        `json:"value"    gencodec:"required"`
    Payload      []byte          `json:"input"    gencodec:"required"`

    // Signature values
    V *big.Int `json:"v" gencodec:"required"`
    R *big.Int `json:"r" gencodec:"required"`
    S *big.Int `json:"s" gencodec:"required"`
}
```

### 交易字段说明

- **AccountNonce**：发送账户的交易次数
- **Price**：Gas价格
- **GasLimit**：Gas限制
- **Recipient**：接收地址，为nil时表示创建合约
- **Amount**：交易金额
- **Payload**：交易数据，对于智能合约调用，包含调用数据
- **V, R, S**：交易签名

```go
func newTransaction(nonce uint64, to *common.Address, amount *big.Int, gasLimit uint64, gasPrice *big.Int, data []byte) *Transaction {
    if len(data) > 0 {
        data = common.CopyBytes(data)
    }
    d := txdata{
        AccountNonce: nonce,
        Recipient:    to,
        Amount:       new(big.Int).Set(amount),
        GasLimit:     gasLimit,
        Price:        new(big.Int).Set(gasPrice),
        Payload:      data,
    }
    return &Transaction{data: d}
}
```

```go
// SignTx signs the transaction with the given private key.
func SignTx(tx *Transaction, s Signer, prv *ecdsa.PrivateKey) (*Transaction, error) {
    h := s.Hash(tx)
    sig, err := crypto.Sign(h[:], prv)
    if err != nil {
        return nil, err
    }
    return tx.WithSignature(s, sig)
}

// WithSignature returns a new transaction with the given signature.
func (tx *Transaction) WithSignature(signer Signer, sig []byte) (*Transaction, error) {
    r, s, v, err := signer.SignatureValues(tx, sig)
    if err != nil {
        return nil, err
    }
    cpy := &Transaction{data: tx.data}
    cpy.data.R, cpy.data.S, cpy.data.V = r, s, v
    return cpy, nil
}
```

```go
// Signer is an interface for transaction signers.
type Signer interface {
    // Sender returns the sender address of the transaction.
    Sender(tx *Transaction) (common.Address, error)
    // SignatureValues returns the signature values to be used when signing the transaction.
    SignatureValues(tx *Transaction, sig []byte) (r, s, v *big.Int, err error)
    // Hash returns the hash of the transaction.
    Hash(tx *Transaction) common.Hash
    // Equal returns true if the signer is equal to the given signer.
    Equal(Signer) bool
}

// EIP155Signer implements Signer using the EIP155 rules.
type EIP155Signer struct {
    chainId, chainIdMul *big.Int
}

// NewEIP155Signer creates a new EIP155Signer.
func NewEIP155Signer(chainId *big.Int) EIP155Signer {
    return EIP155Signer{
        chainId:     chainId,
        chainIdMul:  new(big.Int).Mul(chainId, big.NewInt(2)),
    }
}

// Hash returns the hash of the transaction.
func (s EIP155Signer) Hash(tx *Transaction) common.Hash {
    return rlpHash([]interface{}{
        tx.data.AccountNonce,
        tx.data.Price,
        tx.data.GasLimit,
        tx.data.Recipient,
        tx.data.Amount,
        tx.data.Payload,
        s.chainId,
        uint(0),
        uint(0),
    })
}
```

```go
// ValidateTransaction validates the transaction.
func ValidateTransaction(tx *Transaction, state *StateDB, blockNumber *big.Int, blockTime *big.Int, cfg *ChainConfig) error {
    // Check that the transaction is not empty
    if len(tx.Data()) > 0 && len(tx.Data()) < 4 {
        return ErrTxDataTooShort
    }
    // Check that the transaction has a valid signature
    if _, err := Sender(NewEIP155Signer(tx.ChainId()), tx); err != nil {
        return err
    }
    // Check that the transaction nonce is correct
    if state.GetNonce(tx.From()) != tx.Nonce() {
        return ErrNonceTooLow
    }
    // Check that the transaction gas limit is within bounds
    if tx.Gas() > cfg.MaxGasLimit {
        return ErrGasLimitExceeded
    }
    // Check that the transaction gas price is valid
    if tx.GasPrice().Sign() < 0 {
        return ErrInvalidGasPrice
    }
    // Check that the transaction amount is valid
    if tx.Value().Sign() < 0 {
        return ErrInvalidValue
    }
    // Check that the sender has enough balance
    if state.GetBalance(tx.From()).Cmp(tx.Cost()) < 0 {
        return ErrInsufficientFunds
    }
    return nil
}
```

```go
// ApplyTransaction applies a transaction to the state.
func ApplyTransaction(msg Message, cfg *ChainConfig, bc *BlockChain, state *StateDB, header *types.Header, tx *Transaction, usedGas *uint64, evm *vm.EVM) (*types.Receipt, error) {
    // Create a new context for the transaction
    context := NewEVMContext(msg, header, bc, nil)
    // Create a new EVM
    if evm == nil {
        evm = vm.NewEVM(context, state, cfg, vm.Config{})
    }
    // Execute the transaction
    _, gas, failed, err := ApplyMessage(evm, msg, usedGas)
    if err != nil {
        return nil, err
    }
    // Create a receipt
    receipt := types.NewReceipt(failed, gas)
    receipt.TxHash = tx.Hash()
    receipt.GasUsed = gas
    // If the transaction created a contract, set the contract address
    if msg.To() == nil {
        receipt.ContractAddress = crypto.CreateAddress(msg.From(), tx.Nonce())
    }
    // Add logs to the receipt
    receipt.Logs = state.GetLogs(tx.Hash())
    receipt.BlockHash = header.Hash()
    receipt.BlockNumber = header.Number
    receipt.TransactionIndex = uint(len(bc.GetBlockReceipts(header.Hash())))
    return receipt, nil
}

// ApplyMessage applies a message to the state.
func ApplyMessage(evm *vm.EVM, msg Message, usedGas *uint64) ([]byte, uint64, bool, error) {
    // Check that the message gas limit is within bounds
    if msg.Gas() > evm.GasLimit {
        return nil, 0, false, ErrGasLimitReached
    }
    // Check that the sender has enough balance
    if evm.StateDB.GetBalance(msg.From()).Cmp(msg.Value()) < 0 {
        return nil, 0, false, ErrInsufficientFunds
    }
    // Transfer the value
    evm.StateDB.SubBalance(msg.From(), msg.Value())
    evm.StateDB.AddBalance(*msg.To(), msg.Value())
    // Execute the message
    var (nret []byte
        gas   uint64
        err   error
    )
    if msg.To() == nil {
        // Contract creation
        ret, gas, err = evm.Create(msg.From(), msg.Data(), msg.Gas(), msg.Value())
    } else {
        // Message call
        ret, gas, err = evm.Call(msg.From(), *msg.To(), msg.Data(), msg.Gas(), msg.Value())
    }
    // Update the used gas
    *usedGas += gas
    // Check for errors
    if err != nil {
        return nil, gas, true, err
    }
    return ret, gas, false, nil
}
```

```go
func (pool *TxPool) AddTx(tx *types.Transaction) error {
    return pool.addTx(tx, true)
}

// addTx adds a transaction to the transaction pool.
func (pool *TxPool) addTx(tx *types.Transaction, local bool) error {
    // Validate the transaction
    if err := pool.validateTx(tx, local); err != nil {
        return err
    }
    // Add the transaction to the pool
    pool.mu.Lock()
    defer pool.mu.Unlock()
    // Get the sender address
    from, err := types.Sender(pool.signer, tx)
    if err != nil {
        return err
    }
    // Check if the transaction is already in the pool
    if pool.isTxInPool(from, tx.Nonce()) {
        return fmt.Errorf("transaction already in pool")
    }
    // Add the transaction to the pending or queued pool
    if pool.currentState.GetNonce(from) == tx.Nonce() {
        // Add to pending pool
        if pool.pending[from] == nil {
            pool.pending[from] = make(map[uint64]*types.Transaction)
        }
        pool.pending[from][tx.Nonce()] = tx
    } else {
        // Add to queued pool
        if pool.queued[from] == nil {
            pool.queued[from] = make(map[uint64]*types.Transaction)
        }
        pool.queued[from][tx.Nonce()] = tx
    }
    // Notify listeners
    pool.txFeed.Send(TxPreEvent{tx, local})
    return nil
}
```

```go
// selectTransactions selects transactions from the transaction pool.
func (w *worker) selectTransactions() ([]*types.Transaction, error) {
    // Get pending transactions
    pending := w.eth.TxPool().Pending()
    // Sort transactions by gas price
    var txs types.Transactions
    for _, accTxs := range pending {
        for _, tx := range accTxs {
            txs = append(txs, tx)
        }
    }
    sort.Sort(types.TxByPriceAndNonce{w.currentState, txs})
    // Select transactions
    var selected []*types.Transaction
    var gas uint64
    for _, tx := range txs {
        if gas+tx.Gas() > w.header.GasLimit {
            continue
        }
        selected = append(selected, tx)
        gas += tx.Gas()
    }
    return selected, nil
}
```
