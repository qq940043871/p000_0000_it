## ⚡ 二、核心代码

### 2.1 秒杀令牌

```java
@Service
public class SeckillService {
    
    @Autowired
    private RedisTemplate redisTemplate;
    
    private static final String SECKILL_LUA = 
        "local stock = redis.call('GET', KEYS[1]) " +
        "if stock == false then return -1 end " +  -- 不存在
        "if tonumber(stock) < tonumber(ARGV[1]) then return 0 end " +  -- 库存不足
        "redis.call('DECRBY', KEYS[1], ARGV[1]) " +
        "return 1";  -- 扣减成功
    
    /**
     * 秒杀抢购
     */
    public SeckillResult seckill(Long userId, Long skuId, Long activityId) {
        // 1. 校验活动是否开始
        Activity activity = getActivity(activityId);
        if (System.currentTimeMillis() < activity.getStartTime()) {
            return SeckillResult.notStart();
        }
        if (System.currentTimeMillis() > activity.getEndTime()) {
            return SeckillResult.ended();
        }
        
        // 2. 校验用户是否已购买
        String boughtKey = "seckill:bought:" + activityId + ":" + userId;
        if (redisTemplate.hasKey(boughtKey)) {
            return SeckillResult.alreadyBought();
        }
        
        // 3. Redis预扣库存
        String stockKey = "seckill:stock:" + skuId;
        Long result = redisTemplate.execute(
            new DefaultRedisScript<>(SECKILL_LUA, Long.class),
            Collections.singletonList(stockKey),
            "1"
        );
        
        if (result == null || result == 0) {
            return SeckillResult.soldOut();
        }
        
        // 4. 发送订单创建消息
        mqProducer.send("seckill-order", SeckillOrderMessage.builder()
            .userId(userId)
            .skuId(skuId)
            .activityId(activityId)
            .build());
        
        return SeckillResult.success();
    }
}
```
