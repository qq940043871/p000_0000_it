```
[草稿] ──► [待审核] ──► [审核中] ──► [已签署] ──► [履行中] ──► [已完成]
  │          │           │          │          │           │

```

### 2.3 模板渲染实现

```java
@Service
public class ContractTemplateService {
    
    @Autowired
    private TemplateRepository templateRepo;
    
    @Autowired
    private ContractRepository contractRepo;
    
    @Autowired
    private TemplateRenderEngine renderEngine;
    
    /**
     * 渲染合同
     */
    public ContractRenderResult renderContract(Long contractId, String format) {
        Contract contract = contractRepo.findById(contractId);
        Template template = templateRepo.findById(contract.getTemplateId());
        
        // 1. 收集合同数据
        Map<String, Object> data = collectContractData(contract);
        
        // 2. 替换变量
        String renderedContent = renderEngine.render(template, data);
        
        // 3. 生成文件
        byte[] fileContent;
        if ("pdf".equals(format)) {
            fileContent = pdfGenerator.generate(renderedContent);
        } else {
            fileContent = docGenerator.generate(renderedContent);
        }
        
        return new ContractRenderResult(fileContent, format);
    }
    
    /**
     * 收集合同数据（支持多数据源）
     */
    private Map<String, Object> collectContractData(Contract contract) {
        Map<String, Object> data = new HashMap<>();
        
        // 基础字段
        data.put("contract_no", contract.getContractNo());
        data.put("contract_name", contract.getContractName());
        data.put("amount", contract.getAmount());
        
        // 甲方信息
        Customer partyA = customerService.getById(contract.getPartyAId());
        data.put("party_a_name", partyA.getCompanyName());
        data.put("party_a_address", partyA.getAddress());
        data.put("party_a_contact", partyA.getContactName());
        data.put("party_a_phone", partyA.getPhone());
        
        // 乙方信息
        Supplier partyB = supplierService.getById(contract.getPartyBId());
        data.put("party_b_name", partyB.getCompanyName());
        
        // 订单明细
        List<OrderItem> items = orderService.getItemsByContract(contractId);
        data.put("order_items", items);
        
        // 关联订单汇总
        data.put("total_quantity", items.stream().mapToInt(OrderItem::getQuantity).sum());
        data.put("total_amount", items.stream().mapToDouble(OrderItem::getAmount).sum());
        
        return data;
    }
}
```

## 🔄 三、合同签署流程

### 3.1 电子签章集成

```java
@Service
public class ESignService {
    
    @Autowired
    private ESignProvider eSignProvider;  // 第三方电子签章服务
    
    /**
     * 发起签署
     */
    public SignFlow createSignFlow(Long contractId, List<Signer> signers) {
        Contract contract = contractRepo.findById(contractId);
        
        // 1. 生成签署文档
        ContractRenderResult result = contractTemplateService.renderContract(contractId, "pdf");
        
        // 2. 创建签署流程
        SignFlow flow = eSignProvider.createFlow(
            SignFlowRequest.builder()
                .documentName(contract.getContractName())
                .documentContent(result.getContent())
                .signers(signers)
                .callbackUrl("/api/contract/sign-callback")
                .build()
        );
        
        // 3. 更新合同状态
        contract.setSignFlowId(flow.getFlowId());
        contract.setStatus(ContractStatus.SIGNING);
        contractRepo.save(contract);
        
        // 4. 发送签署通知
        for (Signer signer : signers) {
            notifyService.sendSignNotify(signer, flow);
        }
        
        return flow;
    }
    
    /**
     * 签署回调处理
     */
    @PostMapping("/sign-callback")
    public void handleSignCallback(@RequestBody SignCallbackDTO callback) {
        if (!verifySignature(callback)) {
            return;
        }
        
        Long contractId = Long.parseLong(callback.getBusinessId());
        Contract contract = contractRepo.findById(contractId);
        
        if (callback.getSignStatus() == SignStatus.COMPLETED) {
            // 签署完成，下载签章后的文档
            byte[] signedDoc = eSignProvider.downloadSignedDoc(callback.getFlowId());
            contractAttachmentService.saveSignedDocument(contractId, signedDoc);
            
            contract.setStatus(ContractStatus.SIGNED);
            contract.setSignedAt(LocalDateTime.now());
        } else if (callback.getSignStatus() == SignStatus.REJECTED) {
            contract.setStatus(ContractStatus.REJECTED);
        }
        
        contractRepo.save(contract);
    }
}
```

## 📊 四、合同履约跟踪

### 4.1 履约节点管理

```java
@Service
public class ContractPerformanceService {
    
    /**
     * 根据合同条款生成履约节点
     */
    public List<PerformanceNode> generateNodes(Long contractId) {
        Contract contract = contractRepo.findById(contractId);
        List<ContractClause> clauses = contractClauseRepo.findByContractId(contractId);
        
        List<PerformanceNode> nodes = new ArrayList<>();
        
        for (ContractClause clause : clauses) {
            if ("delivery".equals(clause.getClauseType())) {
                // 交货节点
                nodes.add(createDeliveryNode(contract, clause));
            } else if ("payment".equals(clause.getClauseType())) {
                // 付款节点
                nodes.add(createPaymentNode(contract, clause));
            } else if ("quality".equals(clause.getClauseType())) {
                // 验收节点
                nodes.add(createQualityNode(contract, clause));
            }
        }
        
        return performanceNodeRepo.saveAll(nodes);
    }
    
    /**
     * 履约节点提醒
     */
    @Scheduled(cron = "0 0 9 * * ?")  // 每天9点
    public void sendPerformanceReminders() {
        // 查询今日待履约节点
        List<PerformanceNode> dueNodes = performanceNodeRepo.findDueNodes(
            LocalDate.now(), 
            LocalDate.now().plusDays(7)
        );
        
        for (PerformanceNode node : dueNodes) {
            Contract contract = contractRepo.findById(node.getContractId());
            
            if (node.getNodeType() == NodeType.DELIVERY) {
                // 发货提醒
                notifyService.sendDeliveryReminder(contract, node);
            } else if (node.getNodeType() == NodeType.PAYMENT) {
                // 付款提醒
                notifyService.sendPaymentReminder(contract, node);
            }
        }
    }
}
```

```java
/**
 * 动态表单字段定义
 */
@Data
public class FormField {
    private String fieldCode;      // 字段编码
    private String fieldName;      // 字段名称
    private FieldType fieldType;   // 字段类型
    private boolean required;      // 是否必填
    private String defaultValue;    // 默认值
    private String placeholder;    // 占位文本
    private Map<String, Object> props;  // 扩展属性
    
    // 验证规则
    private List<ValidationRule> validations;
    
    // 联动配置
    private List<LinkageRule> linkages;
}

/**
 * 动态表单渲染
 */
@Service
public class DynamicFormService {
    
    /**
     * 根据表单定义渲染表单
     */
    public String renderFormHtml(FormDefinition definition) {
        StringBuilder html = new StringBuilder();
        
        for (FormField field : definition.getFields()) {
            html.append(renderField(field));
        }
        
        return html.toString();
    }
    
    private String renderField(FormField field) {
        switch (field.getFieldType()) {
            case TEXT:
                return renderTextField(field);
            case SELECT:
                return renderSelectField(field);
            case DATE:
                return renderDateField(field);
            case NUMBER:
                return renderNumberField(field);
            case ATTACHMENT:
                return renderAttachmentField(field);
            default:
                return "";
        }
    }
    
    private String renderTextField(FormField field) {
        return String.format("""
            <div class="form-field">
                <label class="field-label %s">%s</label>
                <input type="text" 
                       name="%s" 
                       class="field-input"
                       placeholder="%s"
                       %s />
            </div>
            """,
            field.isRequired() ? "required" : "",
            field.getFieldName(),
            field.getFieldCode(),
            field.getPlaceholder(),
            field.isRequired() ? "required" : ""
        );
    }
}
```

## ⚠️ 六、常见问题与解决方案

| 问题 | 解决方案 |
|------|----------|
| 不同客户合同格式差异大 | 模板市场+客户自定义模板 |
| 合同签署周期长 | 短信/微信提醒+签署催促 |
| 履约节点遗漏 | 自动生成节点+提前提醒 |
| 合同版本混乱 | 版本控制+变更记录审计 |
| 批量处理效率低 | 批量签署+批量生成 |
