### 3.2 渲染服务实现

```java
@Service
@Slf4j
public class FormRenderService {
    
    @Autowired
    private FieldComponentRegistry componentRegistry;
    
    @Autowired
    private FormDataService formDataService;
    
    /**
     * 渲染表单JSON -> Vue组件
     */
    public String renderFormJson(FormDefinition formDef) {
        StringBuilder vueTemplate = new StringBuilder();
        vueTemplate.append("<el-form ref='form' :model='formData' label-width='120px'>");
        
        for (FieldDefinition field : formDef.getFields()) {
            if (field.getIsHidden()) {
                vueTemplate.append(renderHiddenField(field));
            } else {
                vueTemplate.append(renderVisibleField(field));
            }
        }
        
        vueTemplate.append("</el-form>");
        return vueTemplate.toString();
    }
    
    /**
     * 渲染单个字段
     */
    private String renderVisibleField(FieldDefinition field) {
        FieldComponent component = componentRegistry.get(field.getFieldType());
        if (component == null) {
            return "<!-- Unknown field type: " + field.getFieldType() + " -->";
        }
        
        return component.getRenderer().apply(field);
    }
    
    /**
     * 验证表单数据
     */
    public ValidationResult validateForm(FormDefinition formDef, Map<String, Object> data) {
        ValidationResult result = new ValidationResult();
        result.setValid(true);
        
        for (FieldDefinition field : formDef.getFields()) {
            Object value = data.get(field.getFieldCode());
            
            // 必填校验
            if (field.getIsRequired() && isEmpty(value)) {
                result.setValid(false);
                result.addError(field.getFieldCode(), field.getFieldName() + "不能为空");
                continue;
            }
            
            // 唯一性校验
            if (field.getIsUnique() && !isEmpty(value)) {
                if (!formDataService.checkUnique(formDef.getFormKey(), 
                        field.getFieldCode(), value, field.getFormId())) {
                    result.setValid(false);
                    result.addError(field.getFieldCode(), field.getFieldName() + "已存在");
                }
            }
            
            // 自定义规则校验
            if (!CollectionUtils.isEmpty(field.getValidationRules())) {
                for (ValidationRule rule : field.getValidationRules()) {
                    ValidationRuleResult ruleResult = executeRule(field, value, rule);
                    if (!ruleResult.isPass()) {
                        result.setValid(false);
                        result.addError(field.getFieldCode(), ruleResult.getMessage());
                    }
                }
            }
        }
        
        return result;
    }
}
```

```java
@Component
public class TextFieldRenderer implements FieldRenderer {
    
    @Override
    public String render(FieldDefinition field) {
        StringBuilder sb = new StringBuilder();
        
        sb.append("<el-form-item");
        sb.append(" label='").append(field.getFieldName()).append("'");
        
        if (field.getIsRequired()) {
            sb.append(" required");
        }

        Map<String, Object> style = field.getStyleConfig();
        if (style != null) {
            if (style.containsKey("labelWidth")) {
                sb.append(" label-width='").append(style.get("labelWidth")).append("'");
            }
            if (style.containsKey("width")) {
                sb.append(" class='form-item-width-").append(style.get("width")).append("'");
            }
        }
        
        sb.append(">");
        
        // 渲染输入框
        sb.append("<el-input");
        sb.append(" v-model='formData.").append(field.getFieldCode()).append("'");
        
        Map<String, Object> config = field.getComponentConfig();
        if (config != null) {
            if (config.containsKey("placeholder")) {
                sb.append(" placeholder='").append(config.get("placeholder")).append("'");
            }
            if (config.containsKey("maxlength")) {
                sb.append(" maxlength='").append(config.get("maxlength")).append("'");
            }
            if (config.containsKey("showWordLimit")) {
                sb.append(" show-word-limit");
            }
            if (field.getIsReadonly()) {
                sb.append(" readonly");
            }
        }
        
        sb.append("></el-input>");
        sb.append("</el-form-item>");
        
        return sb.toString();
    }

        if (config != null) {
            if (config.containsKey("placeholder")) {
                sb.append(" placeholder='").append(config.get("placeholder")).append("'");
            }
            if (Boolean.TRUE.equals(config.get("clearable"))) {
                sb.append(" clearable");
            }
            if (Boolean.TRUE.equals(config.get("filterable"))) {
                sb.append(" filterable");
            }
            if (Boolean.TRUE.equals(config.get("multiple"))) {
                sb.append(" multiple collapse-tags");
            }
        }
        
        sb.append(">");
        
        // 渲染选项
        for (Map<String, String> option : options) {
            sb.append("<el-option");
            sb.append(" label='").append(option.get("label")).append("'");
            sb.append(" value='").append(option.get("value")).append("'");
            sb.append(">");
            sb.append("</el-option>");
        }
        
        sb.append("</el-select>");
        sb.append("</el-form-item>");
        
        return sb.toString();
    }
    
    private List<Map<String, String>> loadOptions(FieldDefinition field) {
        String datasourceType = field.getDatasourceType();
        
        if ("static".equals(datasourceType)) {
            // 静态数据
            return field.getComponentConfig().get("options");
        } else if ("dict".equals(datasourceType)) {
            // 字典数据
            String dictCode = field.getDatasourceConfig().get("dictCode");
            return dictionaryService.getItems(dictCode);
        } else if ("api".equals(datasourceType)) {
            // API数据源
            return fetchApiOptions(field);
        }
        
        return Collections.emptyList();
    }
}
```

### 5.2 联动实现

```java
/**
 * 表单联动引擎
 */
@Service
public class FormLinkageEngine {
    
    /**
     * 处理字段变更触发联动
     */
    public List<LinkageAction> handleFieldChange(FormDefinition formDef, 
                                                  String changedField, 
                                                  Object newValue,
                                                  Map<String, Object> formData) {
        List<LinkageAction> actions = new ArrayList<>();
        
        for (LinkageRule rule : formDef.getLinkageRules()) {
            if (rule.getTrigger().getField().equals(changedField)) {
                // 评估触发条件
                if (evaluateCondition(rule, formData)) {
                    actions.addAll(rule.getActions());
                }
            }
        }
        
        return actions;
    }
    
    /**
     * 执行联动动作
     */
    public void executeActions(LinkageAction action, Map<String, Object> formData) {
        switch (action.getActionType()) {
            case "show":
                // 显示字段
                action.setVisible(true);
                break;
                
            case "hide":
                // 隐藏字段
                action.setVisible(false);
                break;
                
            case "setValue":
                // 设置值
                String value = action.getValue();
                if (value.startsWith("${") && value.endsWith("}")) {
                    // 引用其他字段值
                    String refField = value.substring(2, value.length() - 1);
                    formData.put(action.getTargetField(), formData.get(refField));
                } else {
                    formData.put(action.getTargetField(), value);
                }
                break;
                
            case "calculate":
                // 计算
                String formula = action.getFormula();
                Object result = evaluateFormula(formula, formData);
                formData.put(action.getTargetField(), result);
                break;
                
            case "reset":
                // 重置
                formData.put(action.getTargetField(), null);
                break;
                
            case "disable":
                action.setReadonly(true);
                break;
                
            case "enable":
                action.setReadonly(false);
                break;
        }
    }
    
    /**
     * 评估条件
     */
    private boolean evaluateCondition(LinkageRule rule, Map<String, Object> formData) {
        if (rule.getConditions() == null || rule.getConditions().isEmpty()) {
            return true;
        }
        
        for (Condition condition : rule.getConditions()) {
            Object fieldValue = formData.get(condition.getField());
            Object targetValue = condition.getValue();
            
            boolean result;
            switch (condition.getOperator()) {
                case "equals":
                    result = Objects.equals(fieldValue, targetValue);
                    break;
                case "notEquals":
                    result = !Objects.equals(fieldValue, targetValue);
                    break;
                case "greaterThan":
                    result = compareNumbers(fieldValue, targetValue) > 0;
                    break;
                case "lessThan":
                    result = compareNumbers(fieldValue, targetValue) < 0;
                    break;
                case "contains":
                    result = fieldValue != null && fieldValue.toString().contains(targetValue.toString());
                    break;
                case "empty":
                    result = fieldValue == null || "".equals(fieldValue);
                    break;
                default:
                    result = false;
            }
            
            if (!result) return false;
        }
        
        return true;
    }
}
```

### 6.2 表单打印导出

```java
/**
 * 表单打印服务
 */
@Service
public class FormPrintService {
    
    @Autowired
    private FormRenderService formRenderService;
    
    @Autowired
    private FormDataService formDataService;
    
    /**
     * 生成打印页面
     */
    public String generatePrintPage(Long instanceId) {
        FormInstance instance = formDataService.getInstance(instanceId);
        FormDefinition formDef = formRenderService.getDefinition(instance.getFormId());
        
        // 构建打印HTML
        StringBuilder html = new StringBuilder();
        html.append(getPrintHeader());
        html.append(renderPrintContent(formDef, instance));
        html.append(getPrintFooter());
        
        return html.toString();
    }
    
    /**
     * 生成PDF
     */
    public byte[] generatePdf(Long instanceId) {
        String html = generatePrintPage(instanceId);
        
        // 使用wkhtmltopdf或Flying Saucer生成PDF
        return pdfGenerator.convert(html);
    }
    
    /**
     * 渲染打印内容
     */
    private String renderPrintContent(FormDefinition formDef, FormInstance instance) {
        StringBuilder sb = new StringBuilder();
        Map<String, Object> data = instance.getDataJson();
        
        // 打印标题
        sb.append("<h1 class='print-title'>").append(instance.getTitle()).append("</h1>");
        sb.append("<div class='print-meta'>");
        sb.append("<span>单号：").append(instance.getInstanceNo()).append("</span>");
        sb.append("<span>创建时间：").append(instance.getCreatedAt()).append("</span>");
        sb.append("</div>");
        
        // 打印字段
        sb.append("<table class='print-table'>");
        for (FieldDefinition field : formDef.getFields()) {
            if (!field.getIsHidden()) {
                sb.append("<tr>");
                sb.append("<th>").append(field.getFieldName()).append("</th>");
                sb.append("<td colspan='3'>");
                sb.append(formatFieldValue(field, data.get(field.getFieldCode())));
                sb.append("</td>");
                sb.append("</tr>");
            }
        }
        sb.append("</table>");
        
        // 打印签章区域
        sb.append("<div class='print-signatures'>");
        sb.append("<div class='signature-item'>");
        sb.append("<span>制单人：________</span>");
        sb.append("<span>日期：________</span>");
        sb.append("</div>");
        sb.append("</div>");
        
        return sb.toString();
    }
}
```
