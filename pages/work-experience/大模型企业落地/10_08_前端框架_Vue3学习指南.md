## 一、Vue3 概述

### 1.1 为什么选择 Vue3

Vue3 是构建大模型应用前端的首选框架之一：

| 特性 | 说明 |
|------|------|
| 组合式 API | 代码组织更灵活 |
| 性能提升 | 虚拟 DOM 重写，编译时优化 |
| 更好的 TypeScript 支持 | 类型安全 |
| 体积更小 | 相比 Vue2 减少约 30% |
| 生态完善 | Vite、Pinia、Vue Router 配套 |

### 1.2 核心概念

```
Vue3 核心概念：

组件 (Component)
├── 模板 (Template)
├── 逻辑 (Script Setup)
└── 样式 (Style)

响应式系统
├── ref / reactive
├── computed
└── watch

组合式函数 (Composables)
└── 类似 Hooks 的逻辑复用机制
```

## 三、基础语法

### 3.1 组合式 API

```vue
<script setup lang="ts">
import { ref, computed, onMounted } from 'vue'

// 响应式数据
const count = ref(0)
const message = ref('Hello')

// 计算属性
const doubled = computed(() => count.value * 2)

// 方法
function increment() {
  count.value++
}

// 生命周期
onMounted(() => {
  console.log('Component mounted')
})
</script>

<template>
  <div>
    <p>{{ message }} {{ doubled }}</p>
    <button @click="increment">点击</button>
  </div>
</template>
```

### 3.2 reactive 对象

```vue
<script setup lang="ts">
import { reactive, toRefs } from 'vue'

const state = reactive({
  name: '',
  loading: false,
  items: [] as string[]
})

// 解构时保持响应式
const { name, loading, items } = toRefs(state)

// 修改
state.name = 'New Name'
</script>
```

### 3.3 条件渲染与列表

```vue
<template>
  <!-- 条件 -->
  <div v-if="loading">加载中...</div>
  <div v-else>
    <p v-show="show">可见</p>
    
    <!-- 列表 -->
    <ul>
      <li v-for="item in items" :key="item.id">
        {{ item.name }}
      </li>
    </ul>
  </div>
</template>
```

## 四、大模型应用组件

### 4.1 对话组件

```vue
<template>
  <div class="chat-container">
    <!-- 消息列表 -->
    <div class="messages" ref="messagesRef">
      <div
        v-for="msg in messages"
        :key="msg.id"
        :class="['message', msg.role]"
      >
        <div class="avatar">{{ msg.role === 'user' ? 'U' : 'AI' }}</div>
        <div class="content">
          <div class="text" v-html="renderMarkdown(msg.content)"></div>
          <div class="time">{{ msg.time }}</div>
        </div>
      </div>
    </div>
    
    <!-- 输入区 -->
    <div class="input-area">
      <textarea
        v-model="inputText"
        @keydown.enter.exact.prevent="sendMessage"
        placeholder="输入消息..."
        rows="3"
      ></textarea>
      <button @click="sendMessage" :disabled="sending">
        {{ sending ? '发送中...' : '发送' }}
      </button>
    </div>
  </div>
</template>

<script setup lang="ts">
import { ref, nextTick } from 'vue'
import MarkdownIt from 'markdown-it'
import hljs from 'highlight.js'

interface Message {
  id: string
  role: 'user' | 'assistant'
  content: string
  time: string
}

const md = new MarkdownIt({
  highlight: (str, lang) => {
    if (lang && hljs.getLanguage(lang)) {
      return hljs.highlight(str, { language: lang }).value
    }
    return ''
  }
})

const messages = ref<Message[]>([])
const inputText = ref('')
const sending = ref(false)
const messagesRef = ref<HTMLElement>()

function renderMarkdown(content: string) {
  return md.render(content)
}

async function sendMessage() {
  if (!inputText.value.trim() || sending.value) return
  
  const userMsg: Message = {
    id: Date.now().toString(),
    role: 'user',
    content: inputText.value,
    time: new Date().toLocaleTimeString()
  }
  
  messages.value.push(userMsg)
  const userInput = inputText.value
  inputText.value = ''
  sending.value = true
  
  // 滚动到底部
  await nextTick()
  messagesRef.value?.scrollTo(0, messagesRef.value.scrollHeight)
  
  // 调用 API
  try {
    const response = await fetch('/api/chat', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ message: userInput })
    })
    
    const data = await response.json()
    
    messages.value.push({
      id: Date.now().toString(),
      role: 'assistant',
      content: data.content,
      time: new Date().toLocaleTimeString()
    })
  } finally {
    sending.value = false
  }
}
</script>

<style scoped>
.chat-container {
  display: flex;
  flex-direction: column;
  height: 100%;
}

.messages {
  flex: 1;
  overflow-y: auto;
  padding: 16px;
}

.message {
  display: flex;
  margin-bottom: 16px;
}

.message.user {
  flex-direction: row-reverse;
}

.message .content {
  max-width: 70%;
  padding: 12px;
  border-radius: 8px;
}

.message.user .content {
  background: #1890ff;
  color: white;
}

.message.assistant .content {
  background: #f5f5f5;
}

.input-area {
  display: flex;
  gap: 8px;
  padding: 16px;
  border-top: 1px solid #eee;
}

.input-area textarea {
  flex: 1;
  padding: 8px;
  border: 1px solid #ddd;
  border-radius: 4px;
  resize: none;
}
</style>
```

### 4.2 流式输出组件

```vue
<script setup lang="ts">
import { ref } from 'vue'

const fullContent = ref('')
const isStreaming = ref(false)

async function startStream() {
  isStreaming.value = true
  fullContent.value = ''
  
  const response = await fetch('/api/chat/stream', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ message: '你好' })
  })
  
  const reader = response.body?.getReader()
  const decoder = new TextDecoder()
  
  while (reader) {
    const { done, value } = await reader.read()
    if (done) break
    
    const chunk = decoder.decode(value)
    fullContent.value += chunk
  }
  
  isStreaming.value = false
}
</script>
```

## 五、状态管理 (Pinia)

### 5.1 定义 Store

```typescript
// stores/chat.ts
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'

interface Message {
  id: string
  role: 'user' | 'assistant'
  content: string
  timestamp: number
}

export const useChatStore = defineStore('chat', () => {
  // 状态
  const messages = ref<Message[]>([])
  const currentSession = ref<string | null>(null)
  
  // 计算
  const messageCount = computed(() => messages.value.length)
  const lastMessage = computed(() => messages.value.at(-1))
  
  // 方法
  function addMessage(msg: Omit<Message, 'id' | 'timestamp'>) {
    messages.value.push({
      ...msg,
      id: Date.now().toString(),
      timestamp: Date.now()
    })
  }
  
  function clearMessages() {
    messages.value = []
  }
  
  return {
    messages,
    currentSession,
    messageCount,
    lastMessage,
    addMessage,
    clearMessages
  }
})
```

### 5.2 使用 Store

```vue
<script setup lang="ts">
import { useChatStore } from '@/stores/chat'

const chatStore = useChatStore()

// 使用
chatStore.addMessage({
  role: 'user',
  content: '你好'
})
</script>
```

## 六、API 请求封装

### 6.1 Axios 实例

```typescript
// api/request.ts
import axios from 'axios'
import { ElMessage } from 'element-plus'

const request = axios.create({
  baseURL: import.meta.env.VITE_API_BASE,
  timeout: 60000
})

// 请求拦截
request.interceptors.request.use(config => {
  const token = localStorage.getItem('token')
  if (token) {
    config.headers.Authorization = `Bearer ${token}`
  }
  return config
})

// 响应拦截
request.interceptors.response.use(
  response => response.data,
  error => {
    if (error.response?.status === 401) {
      // 登录过期
      ElMessage.error('登录已过期，请重新登录')
      router.push('/login')
    } else {
      ElMessage.error(error.message || '请求失败')
    }
    return Promise.reject(error)
  }
)

export default request
```

### 6.2 API 模块

```typescript
// api/chat.ts
import request from './request'

export const chatApi = {
  // 聊天
  chat(data: { message: string; sessionId?: string }) {
    return request.post('/chat', data)
  },
  
  // 流式聊天
  chatStream(data: { message: string }) {
    return request.post('/chat/stream', data, {
      responseType: 'stream'
    })
  },
  
  // 获取历史
  getHistory(sessionId: string) {
    return request.get(`/chat/history/${sessionId}`)
  }
}

// api/search.ts
export const searchApi = {
  search(data: { query: string; category?: string; topK?: number }) {
    return request.post('/search', data)
  }
}
```

```typescript
  }
```

## 九、最佳实践

### 9.1 组件拆分原则

```
拆分维度：
- 单一职责：每个组件只做一件事
- 可复用性：业务无关的抽离为公共组件
- 组合优于继承：使用 composables 复用逻辑
```

### 9.2 性能优化

```vue
<script setup lang="ts">
// 使用 shallowRef 优化大对象
const largeData = shallowRef([])

// 合理使用 v-memo
<div v-memo="[cacheKey]">
  <!-- 仅在 cacheKey 变化时更新 -->
</div>

// 路由懒加载
const HeavyComponent = defineAsyncComponent(() => 
  import('@/views/Heavy.vue')
)
</script>
```

### 9.3 TypeScript 最佳实践

```typescript
// 定义类型
interface ApiResponse<T> {
  code: number
  data: T
  message: string
}

// 通用响应类型
```
