计算机视觉

计算机视觉是人工智能的一个重要分支，致力于让计算机能够"看"和理解图像和视频内容，实现类似人类视觉的功能。

计算机视觉技术在自动驾驶、医疗影像分析、安防监控、工业检测等领域有广泛应用。

计算机视觉核心任务

图像分类

将图像分配到预定义的类别中，如识别图像中是猫还是狗。

目标检测

不仅识别图像中有什么物体，还要确定它们的位置（边界框）。

图像分割

将图像中的每个像素分配给相应的对象或区域，实现精细的物体识别。

图像生成

生成新的图像内容，如风格迁移、超分辨率重建等。

图像处理基础

OpenCV基础操作

import cv2
import numpy as np
import matplotlib.pyplot as plt

# 读取图像
img = cv2.imread('image.jpg')

# 转换颜色空间（BGR to RGB）
img_rgb = cv2.cvtColor(img, cv2.COLOR_BGR2RGB)

# 调整图像大小
resized = cv2.resize(img_rgb, (224, 224))

# 图像灰度化
gray = cv2.cvtColor(img_rgb, cv2.COLOR_RGB2GRAY)

# 边缘检测
edges = cv2.Canny(gray, 100, 200)

# 显示结果
fig, axes = plt.subplots(2, 2, figsize=(10, 10))
axes[0, 0].imshow(img_rgb)
axes[0, 0].set_title('原始图像')
axes[0, 1].imshow(resized)
axes[0, 1].set_title('调整大小')
axes[1, 0].imshow(gray, cmap='gray')
axes[1, 0].set_title('灰度图像')
axes[1, 1].imshow(edges, cmap='gray')
axes[1, 1].set_title('边缘检测')
plt.show()

卷积神经网络(CNN)基础

import torch
import torch.nn as nn
import torch.nn.functional as F

class SimpleCNN(nn.Module):
    def __init__(self, num_classes=10):
        super(SimpleCNN, self).__init__()
        # 卷积层
        self.conv1 = nn.Conv2d(3, 32, kernel_size=3, padding=1)
        self.conv2 = nn.Conv2d(32, 64, kernel_size=3, padding=1)
        # 池化层
        self.pool = nn.MaxPool2d(2, 2)
        # 全连接层
        self.fc1 = nn.Linear(64 * 8 * 8, 512)
        self.fc2 = nn.Linear(512, num_classes)
        # Dropout
        self.dropout = nn.Dropout(0.5)
        
    def forward(self, x):
        # 第一个卷积块
        x = self.pool(F.relu(self.conv1(x)))  # (32, 16, 16)
        # 第二个卷积块
        x = self.pool(F.relu(self.conv2(x)))  # (64, 8, 8)
        # 展平
        x = x.view(-1, 64 * 8 * 8)
        # 全连接层
        x = F.relu(self.fc1(x))
        x = self.dropout(x)
        x = self.fc2(x)
        return x

# 创建模型
model = SimpleCNN(num_classes=10)
print(model)

CNN通过卷积层提取图像的空间特征，是计算机视觉任务的基础架构。

import torchvision.models as models
import torchvision.transforms as transforms
from PIL import Image
import torch

# 加载预训练的ResNet模型
model = models.resnet50(pretrained=True)
model.eval()  # 设置为评估模式

# 图像预处理
 preprocess = transforms.Compose([
    transforms.Resize(256),
    transforms.CenterCrop(224),
    transforms.ToTensor(),
    transforms.Normalize(mean=[0.485, 0.456, 0.406], std=[0.229, 0.224, 0.225]),
])

# 加载并预处理图像
img = Image.open("test_image.jpg")
input_tensor = preprocess(img)
input_batch = input_tensor.unsqueeze(0)  # 添加batch维度

# 进行预测
with torch.no_grad():
    output = model(input_batch)
    
# 获取预测结果
_, predicted_idx = torch.max(output, 1)
print(f"预测类别索引: {predicted_idx.item()}")

目标检测算法

R-CNN系列

包括R-CNN、Fast R-CNN、Faster R-CNN，通过区域提议网络实现目标检测。

YOLO系列

You Only Look Once，将目标检测作为回归问题，实现端到端的实时检测。

SSD

Single Shot MultiBox Detector，通过不同尺度的特征图检测不同大小的目标。

Transformer检测器

如DETR，使用Transformer架构进行目标检测，无需手工设计的组件。

图像分类服务架构

# Flask + TensorFlow图像分类服务
from flask import Flask, request, jsonify
import tensorflow as tf
import numpy as np
from PIL import Image
import io

app = Flask(__name__)

# 加载预训练模型
model = tf.keras.applications.MobileNetV2(weights='imagenet')

@app.route('/classify', methods=['POST'])
def classify_image():
    # 获取上传的图像
    file = request.files['image']
    image = Image.open(io.BytesIO(file.read()))
    
    # 预处理图像
    image = image.resize((224, 224))
    image_array = tf.keras.preprocessing.image.img_to_array(image)
    image_array = tf.expand_dims(image_array, 0)
    image_array = tf.keras.applications.mobilenet_v2.preprocess_input(image_array)
    
    # 进行预测
    predictions = model.predict(image_array)
    decoded_predictions = tf.keras.applications.mobilenet_v2.decode_predictions(predictions, top=3)[0]
    
    # 返回结果
    results = [{'label': label, 'probability': float(prob)} for (_, label, prob) in decoded_predictions]
    return jsonify(results)

if __name__ == '__main__':
    app.run(host='0.0.0.0', port=5000)

实时目标检测系统

import cv2
import torch
from torchvision import models
from torchvision.models.detection import fasterrcnn_resnet50_fpn

# 加载预训练的目标检测模型
model = fasterrcnn_resnet50_fpn(pretrained=True)
model.eval()

# 打开摄像头
cap = cv2.VideoCapture(0)

while True:
    ret, frame = cap.read()
    if not ret:
        break
    
    # 预处理图像
    img_tensor = torch.from_numpy(frame.transpose((2, 0, 1))).float() / 255.0
    img_tensor = img_tensor.unsqueeze(0)
    
    # 进行目标检测
    with torch.no_grad():
        predictions = model(img_tensor)
    
    # 绘制检测结果
    boxes = predictions[0]['boxes']
    labels = predictions[0]['labels']
    scores = predictions[0]['scores']
    
    for box, label, score in zip(boxes, labels, scores):
        if score > 0.5:  # 置信度阈值
            x1, y1, x2, y2 = box.int().tolist()
            cv2.rectangle(frame, (x1, y1), (x2, y2), (0, 255, 0), 2)
            cv2.putText(frame, f'Object: {score:.2f}', (x1, y1-10), 
                       cv2.FONT_HERSHEY_SIMPLEX, 0.5, (0, 255, 0), 2)
    
    # 显示结果
    cv2.imshow('Object Detection', frame)
    
    if cv2.waitKey(1) & 0xFF == ord('q'):
        break

cap.release()
cv2.destroyAllWindows()
